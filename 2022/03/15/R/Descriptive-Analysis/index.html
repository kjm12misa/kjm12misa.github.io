  • 분산

weight = c(64, 68, 70, 72, 76)
weight_mean = mean(weight)# 평균
weight_deviation = weight - weight_mean # 편차 A - B
weight_deviation
sum(weight_deviation)

weight_deviation_2 = weight_deviation ^2
sum(weight_deviation_2)
mean(weight_deviation_2)

  • 표준 편차

sqrt(mean(weight_deviation_2))

var(weight) - 표본 분산
sd(weight) - 표본 표준 편차

tickers = c(“AAPL”, “TSLA”)
getSymbols(tickers,
from = “2022-01-02”,
to = “2022-01-30”)

stock <- map(tickers, function(x) Ad(get(x)))
stock <- reduce(stock, merge)
colnames(stock) <- tickers

head(stock)

stock_df <- stock %>% data.frame(date = index(stock))
stock_df

  • 사용자 정의 함수

cv_fun <- function(data) {
result = sd(data) / mean(data) * 100
return(result)
}

-> 위와 같은 식을 알고리즘이라고 한다.

cv_fun(stock_df$AAPL)
cv_fun(stock_df$TSLA)

ggplot(stock_df, aes(x = date)) +

  • 사분위수 : 전체 자료를 균등하게 4개의 그룹으로 나눈 값(EX>박스그래프)

  • 이상치판별

qs_df = quantile(albums$sales)
qs_df[4] - qs_df[2]
IQR(albums$sales) -> 3사분위 - 1사분위

boxplot(albums$sales)

  • 이상치 판별

sales2 = c(albums$sales, 450, 460, -100, -1000)

q = quantile(sales2)
boxplot(sales2)

  • 하한 극단 경계값 찾기

bottom_outlier = q[2] - 1.5 * (q[4] - q[2])
bottom_outlier

  • 상한 극단 경계값 찾기
    top_outlier = q[4] + 1.5 * (q[4] - q[2])
    top_outlier

  • 실제 극단값 존재하는지 출력

sales2[sales2 < bottom_outlier]
sales2[sales2 > top_outlier]
